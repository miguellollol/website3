Loop: Creating the main game code 

#include <iostream> 
#include <time.h> 
using namespace std;  

int main() 
{
srand(time(0)); 
int RN = 1 + (rand() % 100); 
int choice; 

//this do loop will go until the player puts the 
//correct number
//it will repeat until the player does so
do 
{
    cout << "Enter a number: ";
    cin >> choice;
    
    if (choice < RN)
    
        cout << "Your estimate is less, than the number ;(" << endl;
        
    else if (choice > RN)
    
        cout << "Your estimate is more, than the secret number :(" << endl;
        
    else
         cout << "Your guess is right!" << endl;
         
} while (choice != RN);

system("PAUSE");
//this pauses the program for flow

 return 0; 
}
